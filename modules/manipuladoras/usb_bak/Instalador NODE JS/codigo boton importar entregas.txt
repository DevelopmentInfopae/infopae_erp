Private Sub Button7_Click(sender As Object, e As EventArgs) Handles BtImpUsb.Click
        Try
            Dim dt As New DataTable
            Dim dt1 As New DataTable

            procesos.consultacampostablacondicion("rutacarpetabackup", "parametros", "id=1", dt1)
            Dim rutac As String = dt1.Rows(0).Item("rutacarpetabackup")

            'MsgBox()

            Dim myProcess As New Process()
            For i As Integer = 1 To 70

                Dim rutabak As String = rutac & "\" & i.ToString.PadLeft(2, "0") & "\BAK.KQ"
                'MsgBox(rutabak)
                If File.Exists(rutabak) Then
                    Dim cadstr As String() = rutac.Split("\")
                    Shell("cmd.exe /k " & String.Format("cd \ & {0} & {1} & {2} & {3} & {4}", cadstr(0), "cd " & cadstr(1), "cd " & cadstr(2), "cd " & cadstr(3), "node convertir" & i.ToString.PadLeft(2, "0") & ".js"), AppWinStyle.Hide)
                    'My.Computer.FileSystem.DeleteFile(rutabak)
                End If
            Next
            MsgBox("Archivos convertidos... Iniciara importacion", MsgBoxStyle.Information)
            For Each foundFile As String In My.Computer.FileSystem.GetFiles(rutac & "\XLS")
                dt.Reset()
                dt.Clear()
                Dim iddisp As Integer = System.IO.Path.GetFileNameWithoutExtension(foundFile).Substring(4, 2)

                Dim cadconex1 As String = "Provider=Microsoft.ACE.OLEDB.12.0;Data Source=" & foundFile & ";Extended Properties='Excel 12.0 Xml;HDR=YES;'"

                Dim cn1 As New OleDbConnection(cadconex1)
                Dim cmd1 As New OleDbCommand
                Dim da1 As New OleDbDataAdapter

                cmd1.Connection = cn1
                cmd1.CommandText = "select * from [Sheet 1$]"
                cmd1.CommandType = CommandType.Text
                da1.SelectCommand = cmd1
                da1.Fill(dt)

                Dim datae As New DataTable

                Dim datas, datas1 As New DataTable
                Dim sum As Integer = 0
                Dim cadsede As String = ""
                For i As Integer = 0 To dt.Rows.Count - 1
                    procesos.consultacampostablacondicion("d.cod_sede,d.id_usuario,e.num_doc", "dispositivos d inner join estudiantes e on d.cod_sede=e.cod_sede", "d.id=" & iddisp & " and e.id_disp_est=" & dt.Rows(i).Item(0), datas)
                    procesos.consultacampostablacondicion("id", "entregas", "fecha_hora like '" & dt.Rows(i).Item(1).ToString.Substring(0, 10) & "%' and id_estudiante='" & datas.Rows(0).Item("num_doc") & "'", datae)
                    procesos.consultacampostablacondicion("nom_sede", "sedes", "cod_sede=" & datas.Rows(0).Item("cod_sede"), datas1)
                    cadsede = datas1.Rows(0).Item("nom_sede")
                    If datae.Rows.Count = 0 Then
                        sum += 1
                        procesos.insertar("entregas", "id_usuario,origen_registro,tipo_dispositivo,id_dispositivo,id_estudiante,fecha_hora,tipo_registro,menu,cod_sede", datas.Rows(0).Item("id_usuario") & ",'USB','Lector de Huellas Dactilar'," & iddisp & ",'" & datas.Rows(0).Item("num_doc") & "','" & dt.Rows(i).Item(1).ToString.Substring(0, 10) & "','Entrega',0," & datas.Rows(0).Item("cod_sede"))
                    End If
                Next
                MsgBox("Importacion Exitosa de " & sum & " Registros de la sede: " & cadsede, MsgBoxStyle.Information)

                My.Computer.FileSystem.DeleteFile(foundFile)
            Next
            MsgBox("Borrando BAK'S...", MsgBoxStyle.Information)
            For i As Integer = 1 To 70
                Dim rutabak As String = rutac & "\" & i.ToString.PadLeft(2, "0") & "\BAK.KQ"
                If File.Exists(rutabak) Then
                    My.Computer.FileSystem.DeleteFile(rutabak)
                End If
            Next
            MsgBox("Proceso Finalizado...", MsgBoxStyle.Information)
        Catch ex As Exception
            MsgBox(ex.Message, MsgBoxStyle.Information)
        End Try
    End Sub
